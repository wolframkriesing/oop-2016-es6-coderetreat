ES 6 Classes: get


"use strict";

class Person {
  constructor(firstName, lastName) {
    this.firstName = firstName;
    this.lastName = lastName;
  }

  get shortedName() {
    return `${this.lastName}, ${this.firstName.charAt(0)}.`;
  }

}

const peter = new Person("Peter", "Smith");
peter.shortedName

ES 6 Classes: set


"use strict";

class Person {
  constructor(firstName, lastName) {
    this.firstName = firstName;
    this.lastName = lastName;
  }
  set fullName(name) {
    this.firstName = name.split(' ')[0];
    this.lastName = name.split(' ')[1];
  }
}

const person1 = new Person("Peter", "Smith");
person1.fullName = 'Steve Wesson';
person1

ES 6 Destructuring: Objects


class Product {
  constructor(name, price) {
    this.name = name;
    this.price = price;
  }
}
let {name, price} = new Product('Tv', 35);
name

ES 6 Destructuring: Params


function formatProduct({name, price, currency = '$'}) {
  return `${name}: ${currency}${price}`;
}
formatProduct({name: 'Tv', price: 35});
